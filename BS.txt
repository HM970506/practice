#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int vect[9] = {1,3,3,4,5,9,15,16,19};
	int input;
	cin >> input;
	int start = 0;
	int end = 8;
	int middle;

	while (1)
	{
		middle = (start + end) / 2;

		if (start >= end) {
			cout << "없음";
			break;
		}

		if (vect[middle] > input) end = middle-1;
		else if (vect[middle] < input) start = middle+1;
		else {
			cout << "찾음";
			break;
		}
		

	}

	return 0;
}
