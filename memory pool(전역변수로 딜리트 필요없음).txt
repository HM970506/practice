
struct Node
{
	int x;
	Node* next;
};

Node* head;
Node* last;

Node buf[1000000];
int bufCnt=0;

Node myAlloc()
{
	return buf[bufCnt++];
}

void addNode(int value)
{
	if (head == NULL)
	{
		head = &myAlloc();
		head->x = value;
		last = head;
	}
	else {
		last->next = &myAlloc();
		last = last->next;
		last->x = value;
	}
}
