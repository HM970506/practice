struct Node
{

	Node* next;
};

Node buf[100000];
int bufcnt = 0;

Node* myAlloc(string value, Node* next)
{
	buf[bufcnt] = { value, next };
	return &buf[bufcnt++];
}

int hashing(string value)
{
	unsigned int result = 0; //21억 이상시 음수방지(42억까지)
	for (int x = 0; x < value.length(); x++)
	{
		result = result * 10 + value[x];
	}
	return result%8;
}

	chain[from] = myAlloc(value, chain[from]);
