#include <iostream>
#include <string>
using namespace std;

struct Node
{
	string a;
	Node* next;
};

Node buf[10];
int bufCnt = 0;

int hashing(string a)
{
	int result = 0;
	int len = a.length();
	for (int x = 0; x < len; x++)
	{
		result =result*10 + (int)(a[x]-'A');
	}
	return result%5;
}

Node* hashcode[5];

Node* myAlloc(string str, Node* next)
{
	buf[bufCnt] = { str, next };
	return &buf[bufCnt++];
}

void addNode(string str, int x)
{
	hashcode[x] = myAlloc(str, hashcode[x]);
	
}

int main()
{
	string map[6] = { "ABS","KBC","TTT","BBS","BHC","ABCD" };

	for (int x = 0; x < 6; x++)
	{
		addNode(map[x], hashing(map[x]));
	}
	string input;
	cin >> input;

	int k = hashing(input);
	for (Node* p = hashcode[k]; p != NULL; p = p->next)
	{
		if (p->a == input)
		{
			cout << "OK";
			return 0;
		}
	}

	cout << "NO";

	return 0;
}
