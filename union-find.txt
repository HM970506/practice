#include <iostream>
using namespace std;

char vect[200]; //숫자를 크게 주는 이유=그룹 이름이 (int)(A)로 들어가니까!

char findBoss(char a) //최종 보스가 나올때까지 반복되는 재귀함수
{
	if (vect[a] == 0) return a; //내가 보스다= 나로 리턴

	char ret = findBoss(vect[a]); //vect[a] = ret, vect[vect[a]]==0 = 최종보스로 리턴
	return ret; 
}
int group(char start, char end) //그룹이 아닌 경우 최종보스를 하나로 통일해 그룹을 맺어줌
{
	char t1 = findBoss(start);
	char t2 = findBoss(end);
	if (t1 == t2) return 0; //둘이 최종보스가 같다

	vect[t2] = t1; //최종보스가 다르면 최종보스를 하나로 묶어준다 
	return 1;
}
int main()
{
	int n;
	cin >> n;
	char start, end;
	char vect[10][2];


	for (int i = 0; i < n; i++) //입력
	{
		cin >> vect[i][0] >> vect[i][1];
	}

	for (int i = 0; i < n; i++) //입력받은 얘들로 사이클 확인
	{
		start = vect[i][0];
		end = vect[i][1];

		if (group(start, end) == 0) //둘이 최종보스가 같다
		{
			cout << "발견"; //발견
			return 0;

		}

	}


	cout << "미발견";


	return 0;
}
