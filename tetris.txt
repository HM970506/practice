void init()
{
	nowx = startx;
	nowy = starty;

	int ret = rand() % 5;

	for (int y = 0; y < 4; y++)
	{
		for (int x = 0; x < 4; x++)
		{
			nowblock[y][x] = block[ret][y][x];
		}
	}
}

void spin()
{
	int afterblock[4][4] = { 0 };

	for (int y = 0; y < 4; y++)
	{
		for (int x = 0; x < 4; x++)
		{
			afterblock[x][3 - y] = nowblock[y][x];
		}
	}

	for (int y = 0; y < 4; y++)
	{
		for (int x = 0; x < 4; x++)
		{
			nowblock[y][x] = 0;
		}
	}

	int minx = 999;
	int miny = 999;
	int maxx = -999;
	int maxy = -999;

	for (int y = 0; y < 4; y++)
	{
		for (int x = 0; x < 4; x++)
		{
			if (afterblock[y][x] == 1)
			{
				if (minx > x)minx = x;
				if (miny > y)miny = y;
				if (maxx < x)maxx = x;
				if (maxy < y)maxy = y;
			}
		}
	}

	int a, b;

	for (int y = miny, a=0; y <= maxy; y++, a++)
	{
		for (int x = minx, b=0; x <= maxx; x++, b++)
		{
			nowblock[a][b] = afterblock[y][x];
		}
	}

}
