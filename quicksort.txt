#include <iostream>
#include <string.h>
using namespace std;

int map[7];

void quicksort(int start, int end)
{
	if (start >= end) return;

	int pivot = start;
	int a = start+1;
	int b = end;

	while (1)
	{
		while (map[a] <= map[pivot] && a<=end) a++;
		while (map[b] > map[pivot] && b>=start) b--;
		if (a > b) break;
		swap(map[a], map[b]);

	}

	swap(map[start], map[b]);

	quicksort(start, b - 1);
	quicksort(b + 1, end);

}

int main()
{
	int n;

	for (int x = 0; x < 7; x++)
	{
		cin >> map[x];
	}
	cin >> n;

	quicksort(0, 6);

	cout << map[n - 1];

	return 0;
}
