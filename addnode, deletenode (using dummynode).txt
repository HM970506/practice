#include <iostream>
using namespace std;

struct Node
{
	char x;
	Node* next;
	Node* before;
};

Node* head;
Node* last;

void push(char value)
{
	if (head == NULL)
	{
		head = new Node();
		head->x = value;
		last = head;
	}
	else
	{
		last->next = new Node();
		last->next->x = value;
		last->next->before = last;
		last = last->next;

	}
}

char pop1()
{
	char backup = head->x;
	head = head->next; //큐
	return backup;
}

char pop2()
{
	char backup = last->x;
	last = last->before;//스택
	return backup;
}

#include <iostream>
using namespace std;

struct Node
{
	int x;
	Node* next;
};

Node *head[6];
Node* last[6];
Node buf[10000000]; //미리 노드 잔뜩 생성
int bufcnt = 0; //사용할 노드 번호

Node *myAlloc(int x, Node* next)
{
	buf[bufcnt].x = x; //만들어둔 노드 사용
	buf[bufcnt].next = next; //헤드의 전 노드=헤드의 다음 노드
	return &buf[bufcnt++];
}

void addNode(int x, int value)
{
	head[x] = myAlloc(value, head[x]);
}
