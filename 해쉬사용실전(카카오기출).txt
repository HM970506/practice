#include <iostream>
#include <string>
#include <unordered_map>
using namespace std;
unordered_multimap<string, string> t;

void init() //모든 경우의 수를 담음
{
	string input[6] = {
		"ABS","ABTC","ABSKG","BCK","BTS","AGKB" };

	for (int x = 0; x < 6; x++)
	{
		string str = "";

		for (int y = 0; y < input[x].length(); y++)
		{
			str += input[x][y];
			t.insert({ str,input[x] });
		}
	}
}
int main()
{
	init();

	string target = "AB*";

	int index = target.find("*");

	if (index != 1)
	{
		target = target.substr(0, target.length() - 1); //*전 값으로 바꾸고
	}

	for (auto i = t.find(target); i !=t.end()  && i->first == target; ++i)
	{
		cout << i->second << " ";
	}
	//i가 end가 아닐때 target해당 해쉬의 모든 칸 cout 

	return 0;
}
