MINHEAP

#include <iostream>
using namespace std;

int heap[200] = { 0 };
int r=1;

void push(int x)
{
	heap[r++] = x; //트리 맨 끝에 x값을 넣는다
	
	int now = r; //맨 끝 층
	int parent;

	while (1)
	{
		parent = now / 2; //부모노드와 비교
		if (now == 1) break; //루트면 끝냄
		if (heap[parent] <= heap[now]) break; //부모가 더 작으면 끝냄

		swap(heap[parent], heap[now]); //부모가 더 크면 스왑함
		now = parent; //한 층 올라감
	}
}

int pop()
{
	int backup = heap[1];
	heap[1] = heap[r];
	heap[r--] = 0;

	int now = 1;
	int son;

	while (1)
	{
		son = now * 2;
		if (son + 1 <= r && heap[son] >= heap[son + 1]) son++; //자식이 둘다 존재시 더 작은 자식 비교
		if (son > r || heap[son] >= heap[now]) break; //자식이 없거나 부모가 더 크면 끝냄

		swap(heap[now], heap[son]); //자식이 더 작으면 스왑함
		now = son; //한 층 내려감
	}

	return backup;
}

int main()
{
	push(3);
	push(5);
	push(2);
	cout<<pop();

	return 0;
}
