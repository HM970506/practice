#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int n;
	cin >> n;

	int bucket[11] = { 0 };
	int input[5] = { 0 };

	for (int x = 0; x < n; x++)
	{
		cin >> input[x];
		bucket[input[x]]++;
	}

	int sum = 0;

	for (int x = 1; x < 11; x++)
	{
		cout << bucket[x-1];
		bucket[x] += bucket[x - 1];
	}


	cout << endl;

	int result[5];

	for (int x = 0; x < 5; x++)
	{
		int index = bucket[input[x]] - 1;
		result[index] = input[x];
		bucket[input[x]]--;
	}

	for (int x = 0; x < 5; x++)
	{
		cout << result[x];
	}

	return 0;
}
