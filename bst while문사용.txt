#include <iostream>
#include <cstring>
using namespace std;

int vect[100] = { 0,15,7,19,0,8,17,25,0,0,0,0,16,18 };
int cnt = 0;

void find(int value)
{
	int now = 1;
	
	while (1)
	{
		if (vect[now] == value)
		{
			cnt++;
			cout << value << ":" << cnt << "회만에찾음" << endl;
			break;
		}

		if (now > 100) {
			cout << value << ":" << "없음" << endl;
			break;
		}

		if (vect[now] > value)
		{
			cnt++;
			now = now * 2;
		}
		else
		{
			cnt++;
			now = now * 2 + 1;
		}
	}

}

int main()
{
	int target[5];

	for (int y = 0; y < 5; y++)
	{
		cin >> target[y];
	}

	for (int x = 0; x < 5; x++)
	{
		find(target[x]);
		cnt = 0;
	}
	return 0;
}
