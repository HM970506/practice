하노이의 탑(한번에 원판 1개만 옮김)

#include <iostream>
using namespace std;

void move(int from, int to)
{
	cout << from << "에서 " << to << "로 옮김" << endl;
}

void hanoi(int n, int from, int by, int to) //탑개수, 1.2.3번탑
{
	if (n == 1) { //1번에 하나 남음
		move(from, to); 
		return;
	}

	else
	{
		hanoi(n - 1, from, to, by); //1->2, 1->3, 1번에 1개남고 끝남
		move(from, to);
		hanoi(n - 1, by, from, to);//
	}
}

int main()
{
	hanoi(3,1,2,3);

	return 0;
}
