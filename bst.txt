#include <iostream>
using namespace std;

int tree[100] = { 0 };

void insert(int x, int now)
{
	if (tree[now] == 0) {
		tree[now] = x;
		return;
	}

	if (tree[now]>=x)insert(x, now*2);
	else insert(x, now*2+1);
}

void search(int x, int now)
{
	if (now >= 100)
	{
		cout << "미발견";
		return;
	}
	if (tree[now] == x) {
		cout << "발견";
		return;
	}

	if (tree[now] >= x) search(x, now * 2);
	else search(x, now * 2 + 1);
}

int main()
{
	insert(4, 1);
	serch(4, 1);

	return 0;
}
